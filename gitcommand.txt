git status                              # aktueller Status 
git log                                 # Log
git clone <url> <where to clone>        # Kopieren einer repository from einem Projekt z.B. GitHub repository

git add file                            # hinzufügen vom File zu zu staging repostoty
git add -A                              # alle Files zu staging directory hinzufügen
git reset file                          # entfernt ein File aus der repository
git reset                               # entfernt alle Dateien aus repository

git commit -m " ..."                    # Commiten zu Repository

git remote -v                           # Viewing Information about remote repository 
git branch -a                           # Viewing Information about branch

git diff				# alle Änderungen anzeigen lassen

git pull origin master
git push origin master

git branch				# gibt alle aktive branches

git branch name-of-branch               # create a branch for the future
git checkout name-of-branch.            # checkout/switch to the branch

git remote add origin https://github.com/abl-xxl/ceres-tst.git # Branch initiieren
